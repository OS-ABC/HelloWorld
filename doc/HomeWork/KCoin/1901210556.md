  [leetcode66.加一](https://leetcode-cn.com/problems/plus-one/)

  ```go
  func plusOne(digits []int) []int {
      var plus int = 1
      var p int = len(digits)-1
      var sum int = 0
      for ;p>=0 && plus>0; {
          sum = digits[p]+plus
          plus = sum / 10
          digits[p] = sum % 10
          p--
      }
      if plus!=0 {
          ans := make([]int, len(digits)+1)
          ans[0]=1
          for i:=1;i<=len(digits);i++ {
              ans[i]=digits[i-1]
          }
          return ans
      }
      return digits
  }
  ```
